## SELECT
```mysql
DINSTINCT 중복 제거

-- Return all the products
-- name
-- unit price
-- new price (unit price * 1.1)
SELECT name, unit_price ,unit_price * 1.1 AS unit_price
FROM products


-- 큰거 보여줌, >, >-, <, <=, =, !=, <>
SELECT *
FROM Customers
WHERE points > 3000


-- Get the orders placed this year

SELECT *
FROM orders
WHERE order_date >= '2019-01-01'


-- AND OR, AND가 더 높음
SELECT *
FROM Customers
WHERE birth_date >= '1990-01-01' AND points > 3000

-- NOT
SELECT *
FROM Customers
WHERE NOT (birth_date >= '1990-01-01' OR points > 1000)


SELECT *
FROM order_items
where order_id = 6 AND unit_price * quantity > 30

-- IN     or 많이 안쓰고 찾을떄 쓰임
SELECT *
FROM Customers
WHERE state IN ('VA', 'FL', 'GA')

-- between
SELECT *
FROM customers
WHERE points BETWEEN 1000 AND 3000



-- LIKE 그것으로 시작하는것 , ~%(시작), %~%(~가있다면),%~(~로 끝)
SELECT *
FROM customers
where last_name LIKE 'b%'



-- _____y 5글자 뛰고 마지막 y 나온다면
SELECT *
FROM customers
WHERE last_name LIKe '_____y'


-- REGEXP   , ^field는 첫번째, fields$는 마지막에 나와야함
SELECT *
FROM customers
WHERE last_name REGEXP 'field$|mac|rose'

-- REGEXP []~ 붙이면 ~있는것중에서 그 앞에 [] 해당하는것들 나옴
-- a-h >> a b c d ~ h
SELECT *
FROM customers
WHERE last_name REGEXP '[a-h]e'

-- ^ beginning
-- $ end
-- | logical or
-- [abcd]
-- [a-f] 


-- b가 포함되어 있는데 뒤에 r이 나오거나 u가 
SELECT *
FROM customers
WHERE last_name regexp 'br|bu'




```
